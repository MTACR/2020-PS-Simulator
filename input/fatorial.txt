// Número para pegar o fatorial está no endereço 100

// Se for menor, bota 1 no endereço do resultado (101)
LOAD 1

RET
// Armazena numero - 1 na posição 100
STORE 100
// Soma 1 no contador
LOAD $102
ADD 1
STORE 102
MULT $101

CALL $46
RET

// Lê o número da entrada
READ 100
